\DoxyCodeLine{00092 \#\# About The Project}
\DoxyCodeLine{00093 <div id="{}about-\/the-\/project"{}></div>}
\DoxyCodeLine{00094 }
\DoxyCodeLine{00095 This projects primary goal is to create a comfortable development environment for working with Markov Models,}
\DoxyCodeLine{00096 as well as creating an end product which can be used for generating password wordlists using Markov Models.}
\DoxyCodeLine{00097 }
\DoxyCodeLine{00098 This project contains following sub-\/projects:}
\DoxyCodeLine{00099 }
\DoxyCodeLine{00100 -\/ MarkovModel}
\DoxyCodeLine{00101   -\/ A versatile header-\/only template library for basic Markov Model structure.}
\DoxyCodeLine{00102 -\/ MarkovAPI}
\DoxyCodeLine{00103   -\/ A static/dynamic library built on MarkovModel, specialized to generate single-\/word lines.}
\DoxyCodeLine{00104 -\/ MarkovAPICLI}
\DoxyCodeLine{00105   -\/ A command line interface built on top of MarkovAPI}
\DoxyCodeLine{00106 -\/ Markopy}
\DoxyCodeLine{00107   -\/ A CPython extension wrapper for MarkovAPI, along with its own command line interface.}
\DoxyCodeLine{00108 -\/ MarkovPasswordsGUI}
\DoxyCodeLine{00109   -\/ A graphical user interface for MarkovAPI}
\DoxyCodeLine{00110 -\/ CudaMarkovAPI}
\DoxyCodeLine{00111   -\/ GPU-\/accelerated wrapper for MarkovAPI}
\DoxyCodeLine{00112 -\/ CudaMarkopy}
\DoxyCodeLine{00113   -\/ GPU-\/accelereted wrapper for CudaMarkovAPI}
\DoxyCodeLine{00114 }
\DoxyCodeLine{00115 }

\DoxyCodeLine{00111 \#\# About The Project}
\DoxyCodeLine{00112 <div id="{}about-\/the-\/project"{}></div>}
\DoxyCodeLine{00113 }
\DoxyCodeLine{00114 This projects primary goal is to create a comfortable development environment for working with Markov Models,}
\DoxyCodeLine{00115 as well as creating an end product which can be used for generating password wordlists using Markov Models.}
\DoxyCodeLine{00116 }
\DoxyCodeLine{00117 This project contains following sub-\/projects:}
\DoxyCodeLine{00118 }
\DoxyCodeLine{00119 -\/ MarkovModel}
\DoxyCodeLine{00120   -\/ A versatile header-\/only template library for basic Markov Model structure.}
\DoxyCodeLine{00121 -\/ MarkovAPI}
\DoxyCodeLine{00122   -\/ A static/dynamic library built on MarkovModel, specialized to generate single-\/word lines.}
\DoxyCodeLine{00123 -\/ MarkovAPICLI}
\DoxyCodeLine{00124   -\/ A command line interface built on top of MarkovAPI}
\DoxyCodeLine{00125 -\/ Markopy}
\DoxyCodeLine{00126   -\/ A CPython extension wrapper for MarkovAPI, along with its own command line interface.}
\DoxyCodeLine{00127 -\/ MarkovPasswordsGUI}
\DoxyCodeLine{00128   -\/ A graphical user interface for MarkovAPI}
\DoxyCodeLine{00129 -\/ CudaMarkovAPI}
\DoxyCodeLine{00130   -\/ GPU-\/accelerated wrapper for MarkovAPI}
\DoxyCodeLine{00131 -\/ CudaMarkopy}
\DoxyCodeLine{00132   -\/ GPU-\/accelereted wrapper for CudaMarkovAPI}
\DoxyCodeLine{00133 }
\DoxyCodeLine{00134 }

\DoxyCodeLine{00180 \#\#\# Using Markopy/CudaMarkopy}
\DoxyCodeLine{00181 <div id="{}using-\/markopy"{}></div>}
\DoxyCodeLine{00182 }
\DoxyCodeLine{00183 You can access basic operations from various model types using the python module, and if you are inexperienced with:}
\DoxyCodeLine{00184 -\/ Libmarkov and Libcudamarkov internals}
\DoxyCodeLine{00185 -\/ C++ code in general}
\DoxyCodeLine{00186 -\/ Importing and extending libraries in general}
\DoxyCodeLine{00187 -\/ Working with Python/C++ intermixed code}
\DoxyCodeLine{00188 }
\DoxyCodeLine{00189 We strongly recommend using the python module.}
\DoxyCodeLine{00190 }
\DoxyCodeLine{00191 While almost all of the python files provide their own entry point, you should use markopy.py or cudamarkopy.py depending on your preferences.}
\DoxyCodeLine{00192 Please note that CUDA code will not run without an NVIDIA graphics card, and without CUDA runtime.}
\DoxyCodeLine{00193 }
\DoxyCodeLine{00194 markopy.py and cudamarkopy.py will let you select the model type you want to use with the -\/mt parameter. With each model, there are slightly different parameters available.}
\DoxyCodeLine{00195 }
\DoxyCodeLine{00196 For top level CLI selector (markopy.py and cudamarkopy.py)}
\DoxyCodeLine{00197 ```}
\DoxyCodeLine{00198 Model Mode selection choices:}
\DoxyCodeLine{00199 usage: cudamarkopy.py [-\/mt MODEL\_TYPE] [-\/h] [-\/ev EVALUATE] [-\/evt EVALUATE\_TYPE]}
\DoxyCodeLine{00200 }
\DoxyCodeLine{00201 Python wrapper for MarkovPasswords.}
\DoxyCodeLine{00202 }
\DoxyCodeLine{00203 optional arguments:}
\DoxyCodeLine{00204   -\/mt MODEL\_TYPE, -\/-\/model\_type MODEL\_TYPE}
\DoxyCodeLine{00205                         Model type to use. Accepted values: MP, MMX}
\DoxyCodeLine{00206   -\/h, -\/-\/help            Model type to use. Accepted values: MP, MMX}
\DoxyCodeLine{00207   -\/ev EVALUATE, -\/-\/evaluate EVALUATE}
\DoxyCodeLine{00208                         Evaluate a models integrity}
\DoxyCodeLine{00209   -\/evt EVALUATE\_TYPE, -\/-\/evaluate\_type EVALUATE\_TYPE}
\DoxyCodeLine{00210                         Evaluation type, model or corpus}
\DoxyCodeLine{00211 }
\DoxyCodeLine{00212         Sample runs:}
\DoxyCodeLine{00213         markopy.py -\/mt MP generate trained.mdl -\/n 500 -\/w output.txt}
\DoxyCodeLine{00214             Import trained.mdl, and generate 500 lines to output.txt}
\DoxyCodeLine{00215 }
\DoxyCodeLine{00216         markopy.py -\/mt MMX generate trained.mdl -\/n 500 -\/w output.txt}
\DoxyCodeLine{00217             Import trained.mdl, and generate 500 lines to output.txt}
\DoxyCodeLine{00218         }
\DoxyCodeLine{00219         cudamarkopy.py -\/mt CUDA generate trained.mdl -\/n 500 -\/w output.txt}
\DoxyCodeLine{00220             Import trained.mdl, and generate 500 lines to output.txt}
\DoxyCodeLine{00221 }
\DoxyCodeLine{00222 ```}

\DoxyCodeLine{00726 Format is `\{left\_node\_content\},weight,\{right\_node\_content\}\(\backslash\)n`}
\DoxyCodeLine{00727 }
\DoxyCodeLine{00728 Example:}
\DoxyCodeLine{00729 }
\DoxyCodeLine{00730 ```}
\DoxyCodeLine{00731 l,5,r}
\DoxyCodeLine{00732 l,12,g}
\DoxyCodeLine{00733 ...}
\DoxyCodeLine{00734 ```}
\DoxyCodeLine{00735 }
\DoxyCodeLine{00736 There are additional requirements for a model for entry and termination nodes.}
\DoxyCodeLine{00737 Entry nodes are represented with 0x00, and termination nodes are represented with 0xff}
\DoxyCodeLine{00738 }
\DoxyCodeLine{00739 It is expected (but not mandatory) to have an edge from starting node to all of the other nodes, and same applies for edges from each nodes to the termination nodes.}
\DoxyCodeLine{00740 }
\DoxyCodeLine{00741 If termination node has any edges that it is position on the left (meaning model file expectation is to traverse to another node after termination node) it will be loaded to the edges, but will be ignored during the random walk logic.}
\DoxyCodeLine{00742 }
\DoxyCodeLine{00743 \#\#\# Corpus}
\DoxyCodeLine{00744 <div id="{}corpus-\/structure"{}></div>}
\DoxyCodeLine{00745 }
\DoxyCodeLine{00746 Corpus files are used to train models. You may chose to train your own model (preferably over an existing base model, like 2gram.mdl) to have your own generation style.}
\DoxyCodeLine{00747 }
\DoxyCodeLine{00748 Corpus file format is:}
\DoxyCodeLine{00749 }
\DoxyCodeLine{00750 `\{occurrence\}\{seperator\}\{string\}\(\backslash\)n`}
\DoxyCodeLine{00751 }
\DoxyCodeLine{00752 -\/ Occurrence is the weight associated with that string. }
\DoxyCodeLine{00753 -\/ Seperator is a single character seperator used to seperate occurrence with the string}
\DoxyCodeLine{00754 -\/ String is the password/string/sequence you want to add to the model.}
\DoxyCodeLine{00755 }
\DoxyCodeLine{00756 }

\DoxyCodeLine{00762 \#\#\#\# Markopy -\/ Python.h -\/ Not found}

\DoxyCodeLine{00772 ```}
\DoxyCodeLine{00773 }
\DoxyCodeLine{00774 and check the shared object filenames. A common issue is that lboost is required but filenames are formatted as libboost, or vice versa.}
\DoxyCodeLine{00775 }
\DoxyCodeLine{00776 Do the same for python related library issues, run: }
\DoxyCodeLine{00777 ```}
\DoxyCodeLine{00778 ls /usr/lib/x86\_64-\/linux-\/gnu/ | grep python}
\DoxyCodeLine{00779 ```}
\DoxyCodeLine{00780 }
\DoxyCodeLine{00781 to verify filename format is as required.}
\DoxyCodeLine{00782 }

\DoxyCodeLine{00786 ```}
